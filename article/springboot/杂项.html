# 全局异常

public class BaseController {

@ExceptionHandler({BaseException.class, Exception.class })
public Response handleException(Exception ex) {
	if (ex instanceof BaseException) {
		BaseException be = (BaseException) ex;
		return new Response(be.getCode(), be.getErrorMessage(), null);
	} else {
		return new Response(ResponseCode.EXCEPTION_NOT_DEFINED, ex.getMessage(), null);
	}
}

}

public class AdminController extends BaseController { @RequestMapping("/exception/undefined") public Response getExceptionUndefined() { return new Response(200, null, 1/1); }

@RequestMapping("/exception/defined")
public void getExceptionDefined() {
	throw new ServerException();
}

}

@SuppressWarnings("serial") public class BaseException extends RuntimeException {

private int code;
private String errorMessage;

public int getCode() {
	return code;
}
public void setCode(int code) {
	this.code = code;
}
public String getErrorMessage() {
	return errorMessage;
}
public void setErrorMessage(String errorMessage) {
	this.errorMessage = errorMessage;
}

}

@SuppressWarnings("serial") public class ServerException extends BaseException {

public ServerException() {
	super();
	setErrorMessage("server exception!");
	setCode(500);
}

}

# profiles

1新建文件application-dev.properties, application-prod.properties 分别加入greeting=hello, dev, greeting=hello, prod

2 在admincontrooler中加入

public @Value("${greeting}") String value; @RequestMapping("/profiles") public String getProfilesValue() { return value; }

3 application.properties中加入spring.profiles.active=dev

4 访问http://localhost:8080/profile 显示hello, dev

5 优先级application-{profile}.properties > application.properties
